/**
     * 新增商品信息
     * 
     * @param commodityInfo 商品信息
     * @return 结果
     */
    @DataChangeLog(table = "商品信息", type = "insert")
    @Override
    public int insertCommodityInfo(CommodityInfo commodityInfo)
    {

        commodityInfoMapper.insertCommodityInfo(commodityInfo);// 插入主表

        // 2. 插入子表
        if ("1".equals(commodityInfo.getMultiUnit()) && commodityInfo.getMultiUnitList() != null) {
            for (CommodityInfoChild child : commodityInfo.getMultiUnitList()) {
                child.setCommodityId(commodityInfo.getCommodityId()); // 设置外键
                String delFlag = "0";
                child.setDelFlag(delFlag);
                child.setCreatedBy(SecurityUtils.getLoginUser().getUser().getNickName());
                child.setCreatedAt(new Date());
                commodityInfoChildMapper.insertCommodityInfoChild(child);
            }
        }

        return 1;

    }

    /**
     * 修改商品信息
     * 
     * @param commodityInfo 商品信息
     * @return 结果
     */
    @DataChangeLog(table = "商品信息", type = "insert")
    @Override
    public int updateCommodityInfo(CommodityInfo commodityInfo)
    {
        // 插入主表
        commodityInfoMapper.updateCommodityInfo(commodityInfo);

        // 删除原子表数据
        commodityInfoChildMapper.deleteCommodityInfoChildByCommodityId(commodityInfo.getCommodityId());

        // 插入新子表数据
        if ("1".equals(commodityInfo.getMultiUnit()) && commodityInfo.getMultiUnitList() != null) {
            for (CommodityInfoChild child : commodityInfo.getMultiUnitList()) {
                child.setCommodityId(commodityInfo.getCommodityId());
                String delFlag = "0";
                child.setDelFlag(delFlag);
                child.setCreatedBy(SecurityUtils.getLoginUser().getUser().getNickName());
                child.setCreatedAt(new Date());
                commodityInfoChildMapper.insertCommodityInfoChild(child);
            }
        }

        return 1;
    }
