下拉框组件本身就可以过滤

```vue
<el-table-column label="检修部位" prop="maintenancePart" width="180">

  <template slot-scope="scope">
    <el-select 
      v-model="scope.row.maintenancePart"
      placeholder="请选择检修部位"
      filterable
      clearable
      @filter-method="(query) => handleFilterChange(query, scope)"
      style="width: 100%"
    >
      <el-option
        v-for="dict in getFilteredOptions(scope)"
        :key="dict.value"
        :label="dict.label"
        :value="dict.value"
      />
    </el-select>
  </template>

</el-table-column>
```

js部分：

```js
data() {
  return {
    // 其他数据...
    maintenancePartSearchMap: new Map() // 用于存储每行的搜索条件
  }
},
methods: {
  // 处理筛选条件变化
  handleFilterChange(query, scope) {
    // 存储每行的搜索条件，使用row的唯一标识作为key
    const rowKey = scope.row.id || scope.$index; // 最好使用row的唯一id
    this.maintenancePartSearchMap.set(rowKey, query);
  },

  // 获取过滤后的选项
  getFilteredOptions(scope) {
    const rowKey = scope.row.id || scope.$index;
    const searchText = this.maintenancePartSearchMap.get(rowKey) || '';
    

    if (!searchText) {
      return this.dict.type.maintenance_part || [];
    }
    
    return (this.dict.type.maintenance_part || []).filter(item => {
      return (
        item.label.toLowerCase().includes(searchText.toLowerCase()) ||
        String(item.value).includes(searchText)
      );
    });

  }
}
```
